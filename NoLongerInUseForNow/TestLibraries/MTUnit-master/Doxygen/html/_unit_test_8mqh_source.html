268 string UnitTest::summary(int count, int countFail)
269 {
270  int countSuccess = count-countFail;
271  double countSuccessPercent;
272  double countFailurePercent;
273  if(count != 0)
274  {
275  countSuccessPercent = 100.0 * countSuccess/count;
276  countFailurePercent = 100.0 * countFail/count;
277  }
278  else
279  {
280  countSuccessPercent = 100.0;
281  countFailurePercent = 0.0;
282  }
283 
284  string s = StringFormat("Total: %d, Success: %d (%.2f%%), Failure: %d (%.2f%%)",
285  count, countSuccess, countSuccessPercent,
286  countFail, countFailurePercent);
287  return(s);
288 }
